
[* This code checks correct guesses and tracks attempts *]
$
    [* NO function definitions *]
$
    integer target, guess, correctGuesses, attempts;
$
    target = 7;
    correctGuesses = 0;
    attempts = 0;
    
    while (correctGuesses < 3) {
        scan(guess);
        print(guess);
        attempts = attempts + 1;

        [* Check if guess is correct *]
        if (guess == target) {
            correctGuesses = correctGuesses + 1;
            print(target);
        } else {
            [* Provide hint based on higher or lower *]
            if (guess < target) {
                print(0);
            } else {
                print(1);
            }
            endif
        }
        endif
        print(attempts);
    }
    endwhile
$
